create table city
(
cityid int not null,
cityname varchar(40),
countrycode char(2),
primary key(cityid)
);
create table weather_type
(
wid int not null,
wtype varchar(40),
primary key(wid)
);
create table observation
(
id int not null,
oid int not null auto_increment,
wid int not null,
risetime varchar(20),
settime varchar(20),
recordtime varchar(20),
humidity int,
preasure float,
primary key(oid),
foreign key(id) references city(cityid),
foreign key(wid) references weather_type(wid)
);
create table wind
(
oid int not null,
speed float,
type varchar(20),
direction varchar(4),
degree float,
foreign key(oid) references observation(oid)
);
create table temperature
(
oid int not null,
val_k float,
val_c float,
foreign key(oid) references observation(oid)
);
